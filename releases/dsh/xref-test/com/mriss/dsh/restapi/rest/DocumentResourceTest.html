88  	@Test
89  	public void testSubmitDocument() throws Exception {
90  
91  		MockMultipartFile firstFile = new MockMultipartFile("contents", "bbc-news-1.pdf", "application/pdf", new FileInputStream(testFile1));
92  
93  		// First request
94  		
95  		startRequest();
96  		
97  		MvcResult result = mockMvc
98  				.perform(
99  						MockMvcRequestBuilders
100 								.multipart("/v1/dsh/document/submit")
101 								.file(firstFile)
102 								.param("title",
103 										"Russia-Trump: FBI chief Wray defends agency"))
104 				.andExpect(status().isOk())
105 				.andExpect(content().contentType(contentType))
106 				.andExpect(
107 						jsonPath("token", org.hamcrest.Matchers.notNullValue()))
108 				.andExpect(
109 						jsonPath("message",
110 								org.hamcrest.Matchers.notNullValue()))
111 				.andReturn();
112 
113 		synchronized (this) {
114 			this.wait(5000);
115 		}
116 		
117 		endRequest();
118 		
119 		String contentAsString = result.getResponse().getContentAsString();
120 		logger.info("Document submission response: " + contentAsString);
121 		TokenDto tDto = new ObjectMapper().readerFor(TokenDto.class).readValue(contentAsString);
122 		Document document = dao.findDocumentByToken(tDto.getToken());
123 		assertNotNull(document);
124 		assertEquals(DocumentStatus.QUEUED_FOR_INDEXING_SUCCESS, document.getDocumentStatus());
125 		
126 		// Second request to test new instances of document submission service and message handler
127 		
128 		firstFile = new MockMultipartFile("contents", "bbc-news-1.pdf", "application/pdf", new FileInputStream(testFile1));

130 		startRequest();

134 						MockMvcRequestBuilders
135 								.multipart("/v1/dsh/document/submit")
136 								.file(firstFile)
137 								.param("title",
138 										"Russia-Trump: FBI chief Wray defends agency"))
139 				.andExpect(status().isOk())
140 				.andExpect(content().contentType(contentType))
141 				.andExpect(
142 						jsonPath("token", org.hamcrest.Matchers.notNullValue()))
143 				.andExpect(
144 						jsonPath("message",
145 								org.hamcrest.Matchers.notNullValue()))
146 				.andReturn();
147 
148 		synchronized (this) {
149 			this.wait(5000);
150 		}
151 		
152 		endRequest();
153 		
154 		contentAsString = result.getResponse().getContentAsString();
155 		logger.info("Document submission response: " + contentAsString);
156 		tDto = new ObjectMapper().readerFor(TokenDto.class).readValue(contentAsString);
