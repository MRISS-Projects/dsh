3   import java.io.InputStream;
4   
5   import com.mriss.dsh.data.models.Document;
6   
7   /**
8    * This service should use other services to handle documents storage (database) and
9    * document data queuing systems. If using fields to store document instances, classes
10   * implementing this interface should be scoped as <code>request</code>.
11   * 
12   * @author riss
13   */
14  public interface DocumentSubmissionService {
15  	
16  	/**
17  	 * Gets a token (new or already used) based from the original contents
18  	 * of a file. The method basically creates a new {@link Document} and
19  	 * return its token for other clients test about processing. The method
20  	 * tests if the document already exists in the database by using a unique
21  	 * hash generated from the original file binary contents. If so, and
22  	 * the flag <code>useCache</code> is set, that old token associated with
23  	 * the found document is returned.
24  	 * @param originalDocumentContents the document original contents 
25  	 * @param useCache if set to <code>true</code>, the method should test if
26  	 * the document already exists in the database using its hash based on its 
27  	 * original binary contents. If document exists and the keywords and sentences document's
28  	 * fields are not empty, then the old token for the old document should be
29  	 * returned. If set to <code>false</code> no test is executed and a new token 
30  	 * for a newly generated document should be returned.
31  	 * @param documentTitle the document title
32  	 * @return the token (new or old) of the document created (or retrieved from database)
33  	 * @see Document
34  	 */
35  	String getTokenFromDocument(InputStream originalDocumentContents, String documentTitle, boolean useCache) throws DocumentSubmissionException;
36  	
37  	/**
38  	 * This method should preferable be implemented to be asynchronous. It should use
39  	 * a previously created document to store it in the documents database and then
40  	 * enqueue the document's data for later processing by other processes. The method
41  	 * should also test if an instance of {@link Document} has the sentences and keywords
42  	 * collections not empty. If they are not empty, and cache settings are set,
43  	 * the method should do nothing.
44  	 * @see Document
45  	 */
46  	void storeDocumentAndQueueForProcessing() throws DocumentSubmissionException ;
47  	
48  }
