3   import org.springframework.beans.factory.annotation.Autowired;
4   import org.springframework.beans.factory.annotation.Qualifier;
5   import org.springframework.integration.annotation.ServiceActivator;
6   import org.springframework.messaging.Message;
7   import org.springframework.messaging.MessageChannel;
8   import org.springframework.messaging.MessageHandler;
9   import org.springframework.messaging.SubscribableChannel;
10  import org.springframework.messaging.support.GenericMessage;
11  import org.springframework.stereotype.Service;
12  
13  @Service
14  public class DocumentQueueServiceImpl implements DocumentQueueService {
15  	
16  	@Autowired
17  	@Qualifier("toRabbit")
18  	private MessageChannel rabbitChannel;
19  
20  	@Autowired
21  	@Qualifier("toRabbitResponse")
22  	private SubscribableChannel toRabbitResponse;
23  
24  	@Override
25  	public void enqueueDocumentId(String documentId, MessageHandler messageHandler) {
26  		this.toRabbitResponse.subscribe(messageHandler);
27  		rabbitChannel.send(new GenericMessage<String>(documentId));
28  	}
29  	
30  	@ServiceActivator (inputChannel = "toRabbitResponse", outputChannel="loggingChannel")
31  	public Message<String> processEnqueueingResponse(String payload) {
32  		synchronized (this) {
33  			this.notifyAll();
34  		}
35  		return new GenericMessage<String>(payload);
36  	}
37  
38  }
