6   
7   import org.apache.commons.lang3.StringUtils;
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;

11  import org.springframework.http.MediaType;
12  import org.springframework.web.bind.annotation.PathVariable;
13  import org.springframework.web.bind.annotation.RequestMapping;
14  import org.springframework.web.bind.annotation.RequestMethod;
15  import org.springframework.web.bind.annotation.RequestParam;
16  import org.springframework.web.bind.annotation.ResponseBody;
17  import org.springframework.web.bind.annotation.RestController;
18  import org.springframework.web.multipart.MultipartFile;
19  
20  import com.mriss.dsh.data.models.Document;
21  import com.mriss.dsh.restapi.dto.DocumentStatusDto;
22  import com.mriss.dsh.restapi.dto.TokenDto;
23  import com.mriss.dsh.restapi.service.DocumentHandlingService;
24  import com.mriss.dsh.restapi.service.DocumentSubmissionException;
25  import com.mriss.dsh.restapi.service.DocumentSubmissionService;
26  
27  @RestController
28  @RequestMapping("/v1/dsh/document")
29  public class DocumentResource {
30  	
31  	public static final String TOKEN_NOT_FOUND_MESSAGE = "No token was found with the contents: ";
32  
33  	public static final String TOKEN_NOT_FOUND = "TOKEN_NOT_FOUND";
34  
35  	private final static Logger log = LoggerFactory.getLogger(DocumentResource.class);
36  	
37  	@Autowired
38  	DocumentSubmissionService documentSubmissionService;
39  	
40  	@Autowired
41  	DocumentHandlingService documentHandlingService;
42  
43  	@RequestMapping(value = "/submit", method = RequestMethod.POST, produces = MediaType.APPLICATION_JSON_VALUE)
44      public @ResponseBody TokenDto submitDocument(HttpServletRequest request, @RequestParam("title") String title, @RequestParam("contents") MultipartFile contents)
45              throws IllegalStateException, IOException {
46  		try {			
47  			validateParameters(title, contents);
48  			String token = documentSubmissionService.getTokenFromDocument(contents.getInputStream(), title, false);
49  			documentSubmissionService.storeDocumentAndQueueForProcessing();
50  			return new TokenDto(token, "");
51  		} catch (Exception e) {
52  			log.error(e.getMessage(), e);
53  			return new TokenDto("ERROR", "Error submitting file: " + e.getMessage()); 
54  		}
55  	}
56  	
57  	@RequestMapping(value = "/status/{token}", method = RequestMethod.GET, produces = MediaType.APPLICATION_JSON_VALUE)
58  	public @ResponseBody DocumentStatusDto getStatus(HttpServletRequest request, @PathVariable("token") String token) {
59  		Document d = documentHandlingService.getDocumentByToken(token);
60  		if (d == null) {
61  			return new DocumentStatusDto(TOKEN_NOT_FOUND, TOKEN_NOT_FOUND_MESSAGE + token);
62  		} else {
63  			return new DocumentStatusDto(d.getDocumentStatus().getStatusDescription(), d.getDocumentStatusMessage());
64  		}
65  	}
66  
67  	private void validateParameters(String title, MultipartFile contents)
68  			throws DocumentSubmissionException {
69  		log.info("Document title: " + title);
70  		if (StringUtils.isBlank(title) || contents == null) {
71  			throw new DocumentSubmissionException("Document title and contents can't be null.");
72  		}
73  	}
74  
75  }
