1   package com.mriss.dsh.restapi.service;

3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   import org.springframework.beans.factory.annotation.Autowired;
6   import org.springframework.beans.factory.annotation.Qualifier;
7   import org.springframework.context.annotation.Scope;
8   import org.springframework.messaging.Message;
9   import org.springframework.messaging.MessagingException;
10  import org.springframework.messaging.SubscribableChannel;
11  import org.springframework.stereotype.Component;
12  
13  import com.mriss.dsh.data.models.Document;
14  import com.mriss.dsh.data.models.TransitionType;
15  
16  @Component
17  @Scope(scopeName="prototype")
18  public class DocumentEnqueueResponseMessageHandler implements DocumentEnqueueMessageHandler {
19  		
20  	final static Logger logger = LoggerFactory.getLogger(DocumentEnqueueResponseMessageHandler.class);
21  	
22  	@Autowired(required = true)
23  	private DocumentHandlingService docHandlingService;
24  	
25  	@Autowired
26  	@Qualifier("toRabbitResponse")
27  	private SubscribableChannel channel;
28  	
29  	private Document submittedDocument;
30  
31  	@Override
32  	public void handleMessage(Message<?> message) throws MessagingException {
33  		logger.info("DocumentEnqueueResponseMessageHandler.this(): " + this);
34  		logger.info("Processing enqueue process response for message: " + message + ". Document: " + submittedDocument);
35  		if (isOk(message)) {
36  			submittedDocument.transitionStatus(TransitionType.SUCCESS);
37  		} else {
38  			submittedDocument.transitionStatus(TransitionType.ERROR);
39  		}
40  		if (docHandlingService != null) docHandlingService.storeDocument(submittedDocument);
41  		if (this.channel != null) this.channel.unsubscribe(this);
42  	}
43  
44  	boolean isOk(Message<?> message) {
45  		return ((String) message.getPayload()).indexOf("sent ok") != -1;
46  	}
47  
48  	@Override
49  	public void setDocument(Document document) {
50  		logger.info("MessageHandler: " + this);
51  		this.submittedDocument = document;
52  	}
53  
54  	public void setDocHandlingService(DocumentHandlingService docHandlingService) {
55  		this.docHandlingService = docHandlingService;
56  	}
57  
58  	
59  }
