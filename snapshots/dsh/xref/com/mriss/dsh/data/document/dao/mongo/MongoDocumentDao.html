3   import java.util.List;
4   import java.util.Optional;
5   
6   import org.springframework.beans.factory.annotation.Autowired;
7   import org.springframework.stereotype.Component;
8   
9   import com.mriss.dsh.data.document.dao.DocumentDao;
10  import com.mriss.dsh.data.models.Document;
11  
12  @Component
13  public class MongoDocumentDao implements DocumentDao {
14  	
15  	@Autowired
16  	DocumentRepository repository;
17  
18  	@Override
19  	public String storeDocument(Document document) {
20  		Document d = repository.save(document);
21  		if (!d.equals(Optional.empty()))
22  			return d.getId();
23  		else 
24  			return null;
25  	}
26  
27  	@Override
28  	public Document findDocumentById(String id) {
29  		Optional<Document> d = repository.findById(id);
30  		if (!d.equals(Optional.empty()))
31  			return d.get();
32  		else 
33  			return null;
34  	}
35  
36  	@Override
37  	public List<Document> findDocumentsByHash(String documentHash) {
38  		return repository.findByFileHash(documentHash);
39  	}
40  
41  	@Override
42  	public Document findDocumentByToken(String token) {
43  		return repository.findByToken(token);
44  	}
45  
46  	@Override
47  	public void clearDocuments() {
48  		repository.deleteAll();
49  	}
50  
51  }
