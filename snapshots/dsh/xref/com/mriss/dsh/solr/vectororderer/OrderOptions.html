3   /**
4    * Class holding the order options. Mainly the {@link Order} instance and the
5    * field used to order.
6    * 
7    * @author riss
8    *
9    */
10  public class OrderOptions {
11  
12      /**
13       * The order type (asc or desc)
14       */
15      private Order order;
16  
17      /**
18       * The field/terms vector param to be used to order.
19       */
20      private String termVectorParam;
21  
22      /**
23       * Default constructor.
24       * 
25       * @param order      the order (asc or desc)
26       * @param orderField the field to be used when ordering.
27       */
28      public OrderOptions(Order order, String orderField) {
29          super();
30          this.order = order;
31          this.termVectorParam = orderField;
32      }
33  
34      /**
35       * Gets the order.
36       * 
37       * @return the order.
38       */
39      public Order getOrder() {
40          return order;
41      }
42  
43      /**
44       * Gets the param/field.
45       * 
46       * @return the param/field.
47       */
48      public String getTermVectorParam() {
49          return termVectorParam;
50      }
51  
52      @Override
53      public String toString() {
54          return "OrderOptions [order=" + order + ", orderField=" + termVectorParam + "]";
55      }
56  
57  }
