1   package com.mriss.dsh.solr.numberfilter;
2   
3   import java.io.IOException;
4   import java.util.regex.Matcher;
5   import java.util.regex.Pattern;
6   
7   import org.apache.lucene.analysis.FilteringTokenFilter;
8   import org.apache.lucene.analysis.TokenStream;
9   import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  public class AdvancedNumberFilter extends FilteringTokenFilter {
14  
15      public static final String MIXED_NUMBER_1 = "(\\D*)\\-(\\d*)";
16      public static final String MIXED_NUMBER_2 = "(\\d*)\\-(\\D*)";
17  
18      Logger LOGGER = LoggerFactory.getLogger(AdvancedNumberFilter.class);
19  
20      private CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
21  
22      protected AdvancedNumberFilter(TokenStream input) {
23          super(input);
24      }
25  
26      @Override
27      protected boolean accept() throws IOException {
28          int length = termAtt.length();
29          char[] buffer = termAtt.buffer();
30          String contents = new String(buffer, 0, length);
31          return validate(contents);
32      }
33  
34      private boolean validate(String contents) {
35          return !isUrl(contents) && !isMixedNumber(contents) && !isHostName(contents);
36      }
37  
38      private boolean isHostName(String contents) {
39          Pattern r1 = Pattern.compile(
40                  "^(([a-zA-Z0-9]|[a-zA-Z0-9][a-zA-Z0-9\\-]*[a-zA-Z0-9])\\.)*([A-Za-z0-9]|[A-Za-z0-9][A-Za-z0-9\\-]*[A-Za-z0-9])$");
41          Matcher m1 = r1.matcher(contents);
42          return contents.contains(".") && m1.matches();
43      }
44  
45      private boolean isMixedNumber(String contents) {
46          Pattern r1 = Pattern.compile(MIXED_NUMBER_1);
47          Matcher m1 = r1.matcher(contents);
48          Pattern r2 = Pattern.compile(MIXED_NUMBER_2);
49          Matcher m2 = r2.matcher(contents);
50          return m1.matches() || m2.matches();
51      }
52  
53      private boolean isUrl(String contents) {
54          Pattern r1 = Pattern.compile("^(https?|ftp|file)://[-a-zA-Z0-9+&@#/%?=~_|!:,.;]*[-a-zA-Z0-9+&@#/%=~_|]");
55          Matcher m1 = r1.matcher(contents);
56          return m1.matches();
57      }
58  }
